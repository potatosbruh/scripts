function easyTP(placeCFrame)
	local player = game.Players.LocalPlayer;
	if player.Character then
		player.Character.HumanoidRootPart.CFrame = placeCFrame
	end
end

function oilCup()
	easyTP(game:GetService("Workspace").oil.Part.CFrame)
end

function bloodCup()
	easyTP(game:GetService("Workspace").blood.Handle.CFrame)
end

function mystCup()
	easyTP(game:GetService("Workspace")["mysterious cup"].Handle.CFrame)
end

function tpForge()
    easyTP(game:GetService("Workspace").Forging.Part.CFrame)
end
-- init
local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/GreenDeno/Venyx-UI-Library/main/source.lua"))()
local venyx = library.new("World of Trollge", 5013109572)

-- themes
local themes = {
Background = Color3.fromRGB(24, 24, 24),
Glow = Color3.fromRGB(0, 0, 0),
Accent = Color3.fromRGB(10, 10, 10),
LightContrast = Color3.fromRGB(20, 20, 20),
DarkContrast = Color3.fromRGB(14, 14, 14),  
TextColor = Color3.fromRGB(255, 255, 255)
}

-- first page
local page = venyx:addPage("Everything", 5012544693)
local sec1 = page:addSection("Autos")
local sec2 = page:addSection("TP's")

sec1:addToggle("oil", nil, function(value)
	while value == true do
		oilCup()
		wait(.5)
	end
end)

sec1:addToggle("blood", nil, function(value)
	while value == true do
		bloodCup()
		wait(.5)
	end
end)

sec1:addToggle("mysterious cup", nil, function(value)
	while value == true do
		mystCup()
		wait(.5)
	end
end)

sec2:addButton("Forge", function()
    tpForge()
end)
